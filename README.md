    <script>
        let tiles = [];
        let moves = 0;
        let timerInterval;
        let seconds = 0;
        let gameActive = false;

        const boardElement = document.getElementById('board');
        const moveDisplay = document.getElementById('move-count');
        const timerDisplay = document.getElementById('timer');
        const winOverlay = document.getElementById('win-message');
        const finalStats = document.getElementById('final-stats');

        function initGame() {
            // Reset state
            tiles = [1, 2, 3, 4, 5, 6, 7, 8, null]; // null is the empty space
            moves = 0;
            seconds = 0;
            gameActive = true;
            moveDisplay.innerText = moves;
            timerDisplay.innerText = "00:00";
            winOverlay.style.display = 'none';
            
            clearInterval(timerInterval);
            startTimer();
            
            shuffleTiles();
            renderBoard();
        }

        function shuffleTiles() {
            // To ensure a solvable puzzle, we perform random valid moves from a solved state
            let shuffleMoves = 100;
            let emptyIdx = tiles.indexOf(null);

            for (let i = 0; i < shuffleMoves; i++) {
                const neighbors = getNeighbors(emptyIdx);
                const randomNeighbor = neighbors[Math.floor(Math.random() * neighbors.length)];
                
                // Swap
                [tiles[emptyIdx], tiles[randomNeighbor]] = [tiles[randomNeighbor], tiles[emptyIdx]];
                emptyIdx = randomNeighbor;
            }
        }

        function getNeighbors(idx) {
            const neighbors = [];
            const row = Math.floor(idx / 3);
            const col = idx % 3;

            if (row > 0) neighbors.push(idx - 3); // Top
            if (row < 2) neighbors.push(idx + 3); // Bottom
            if (col > 0) neighbors.push(idx - 1); // Left
            if (col < 2) neighbors.push(idx + 1); // Right

            return neighbors;
        }

        function renderBoard() {
            boardElement.innerHTML = '';
            tiles.forEach((number, index) => {
                const tileDiv = document.createElement('div');
                tileDiv.className = `tile ${number === null ? 'empty' : ''}`;
                tileDiv.innerText = number || '';
                
                if (number !== null) {
                    tileDiv.addEventListener('click', () => handleTileClick(index));
                }
                
                boardElement.appendChild(tileDiv);
            });
        }

        function handleTileClick(index) {
            if (!gameActive) return;

            const emptyIdx = tiles.indexOf(null);
            const neighbors = getNeighbors(index);

            if (neighbors.includes(emptyIdx)) {
                // Swap
                [tiles[index], tiles[emptyIdx]] = [tiles[emptyIdx], tiles[index]];
                moves++;
                moveDisplay.innerText = moves;
                renderBoard();
                checkWin();
            }
        }

        function checkWin() {
            const winState = [1, 2, 3, 4, 5, 6, 7, 8, null];
            const isWin = tiles.every((val, index) => val === winState[index]);

            if (isWin) {
                gameActive = false;
                clearInterval(timerInterval);
                finalStats.innerText = `Finished in ${moves} moves and ${timerDisplay.innerText}!`;
                winOverlay.style.display = 'flex';
                winOverlay.classList.add('animate-win');
            }
        }

        function startTimer() {
            timerInterval = setInterval(() => {
                seconds++;
                const mins = Math.floor(seconds / 60).toString().padStart(2, '0');
                const secs = (seconds % 60).toString().padStart(2, '0');
                timerDisplay.innerText = `${mins}:${secs}`;
            }, 1000);
        }

        // Initialize on load
        window.onload = initGame;
    </script>
